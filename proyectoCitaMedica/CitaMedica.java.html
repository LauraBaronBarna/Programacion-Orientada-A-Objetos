import java.util.Date;

public class CitaMedica
{
    // Inicializar los atributos de la clase
    private String paciente;
    private Date fecha;
    private String motivo;

    // Constructor de la clase
    public CitaMedica(String paciente, Date fecha, String motivo) {
        this.paciente = paciente;
        this.fecha = fecha;
        this.motivo = motivo;
    }

    // Método para obtener info de la cita medica
    public String obtenerInformacionCita() {
        return "Paciente: " + paciente + "\nFecha: " + fecha + "\nMotivo: " + motivo;
    }

    // Método para crear una cita medica desde la consola
    public static CitaMedica crearCitaDesdeConsola() throws ParseException
    {
        // Formato de fecha para la entrada del usuario
        SimpleDateFormat dateFormat = new SimpleDateFormat("dd/MM/yyyy HH:mm");
        // Solicitar el nombre del paciente
        System.out.println("Ingrese el nombre del paciente: ");
        String paciente = new Scanner(System.in).nextLine();

        //Solicitar la fecha de la cita
        System.out.println("Ingrese la fecha de la cita: (formato dd/MM/yyyy HH:mm)");
        Date fecha = dateFormat.parse(new Scanner(System.in).nextLine());

        //Solicitar el motivo de la cita
        System.out.println("Ingrese el motivo de la cita: ");
        String motivo = new Scanner(System.in).nextLine();

        //Crear y devolver una nueva instancia de CitaMedica con los datos proporcionados por el usuario
        return new CitaMedica(paciente, fecha, motivo);
    }
}
